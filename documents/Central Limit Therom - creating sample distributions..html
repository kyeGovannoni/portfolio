#This class when called with generate a rnd number between 0 and the length of the sample list.
#It will loop this process s_len times to create a sample, this will be appended to n_samples.
#This whole process will look n times to generate a n amount of randomly selected samples from the sample data. 

def Fetch_samples(data,n=None,sample_size=None):
    if n == None:
        n=10    
   
    if sample_size == None:
        sample_size =5 
        
    n_samples=[]
    for i in range(0,sample_size):
        s = []
        for i in range(0,n):
            i = random.randint(0,(len(data)-1))
            s.append(data[i])
        n_samples.append(s)
    
    print(len(n_samples),"Random Samples taken") 
    
    return n_samples

def exponential_dist(x_lim, r=None, p0=None, decimal=None):
    if decimal ==None:
        decimal = 10
    if r == None:
        r=1 
    if p0 == None:
        p0=1
        
    x_list = []
    f_list = []
    
    for i in range(0,(x_lim*decimal)):
        t = i/decimal 
        x_list.append(t)
        f_list.append(p0*np.exp(r*t))


    
    return x_list,f_list

f(x) = 1*e^rt

This class's purpose is to populate a list with the distribution data, with variables to set the boundaries.

def Fetch_exp_samples(x,p,n=None,s_size=None,st = None,print_info=True):
    if n == None:
        n=5
    if s_size ==None:
        s_size = 10    
    if st == None:
        show = True 
    else: 
        show = False
  
    start_range = x.iloc[0]
    step = round(x.iloc[len(x)-1] - x.iloc[len(x)-2],2)
    end_range = x.iloc[len(x)-1] + step
    
    samples=[]
   
    for i in range(0,s_size):
        s = []
        for i in range(0,n):
            s.append(np.random.choice(np.arange(start_range,end_range,step), p=p))
        
        samples.append(s)
    if print_info == True:
        print('x start',start_range)
        print('x stop',end_range)
        print('x step',step)
        print('\n')
        print(s_size,'Samples Fetched')
        print('n:',n)
    
    return samples

def Fetch_dist_data(sample_list,show_table = None,plot=True,print_info=True):#from Fetch_exp_samples. 
    if show_table == None:
        show = True
    else:
        show = False
   
    s = pd.DataFrame()
    s = pd.DataFrame(sample_list)
    s['sample_means'] = s.mean(axis=1)
    
    
    dist_data = s['sample_means'].value_counts().rename_axis('s_mean').reset_index(name='count') 
    #losing rows? 
    dist_data['val*f(x)'] = dist_data['s_mean']*dist_data['count']
    
    s_distribution_mean = dist_data['val*f(x)'].sum()/len(s)   
    
    dist_data['(x-u)^2'] = (dist_data['s_mean'] - s_distribution_mean)**2
    dist_data['(x-u)^2 *f(x)'] =  dist_data['(x-u)^2'] *dist_data['count']
    variance = (dist_data['(x-u)^2 *f(x)'].sum())/len(s)
    std_d = math.sqrt(variance)
    
    if print_info == True:
        print('Mean of sample means:',round(s_distribution_mean,2))
        print('Standard error of the mean:',round(std_d,2))
    
    
    if plot == True:    
        fig = plt.figure(figsize=(5,3))
        sns.histplot(s['sample_means'])
        plt.show()
        
    if show == True:
        return dist_data
    elif show ==False:
        return

Confirming standard error of the mean¶

\begin{equation*} \underset{\bar{X}}{\sigma^2} = \frac{\sigma^2_\mu}{n} \space\space\space\space or \space\space\space\space \sigma_\bar{X} = \frac{\sigma_\mu}{\sqrt{n}} \end{equation*}
